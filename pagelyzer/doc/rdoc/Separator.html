     # File lib/pagelyzer_separator.rb, line 121
121:         def process_weight(mode)
122:                 #* assign a weight to a separator
123:                 #* greater distance greater weight
124:                 #* different types, font, background of content in both sides of separator higher weight
125:                 max_set = []
126:                 @bright_side.each do |b|
127:                         max_dist = 0
128:                         @dark_side.each do |d|
129:                                 if mode==:horizontal
130:                                         dist = Point.new(0,b.max_y).distance_to? Point.new(0,d.min_y)
131:                                 else
132:                                         dist = Point.new(b.max_x,0).distance_to? Point.new(d.min_x,0)
133:                                 end
134:                                 if dist > max_dist
135:                                         max_dist = dist
136:                                 end
137:                         end
138:                         max_set.push max_dist
139:                 end
140:                 @weight = max_set.max unless max_set.empty?
141:                 
142:                 # backgrounds differents
143:                 bg = @bright_side.collect {|x| x.candidates_background_color_list}.flatten.uniq & @dark_side.collect {|x| x.candidates_background_color_list}.flatten.uniq
144:                 @weight += 10 if bg==[]
145:                 
146:                 #structures of blocks of two sides are are very similar (text, images, etc) decrease weight
147:                 a=@bright_side.collect {|x| x.candidates_structure}.flatten.uniq.first
148:                 b=@dark_side.collect {|x| x.candidates_structure}.flatten.uniq.first
149:                 @weight -=10 if a=='TITLE' and b!='TITLE'
150:                 @weight -=10 if a==b and a!='COMBINED'
151:                 
152:                 #fontsizes if different increase weight
153:                 a=@bright_side.collect {|x| x.candidates_font_size_list}.flatten.max
154:                 b=@dark_side.collect {|x| x.candidates_font_size_list}.flatten.min
155:                 a=0 if a.nil?
156:                 b=0 if b.nil?
157:                 @weight +=10 if a < b
158:         end
